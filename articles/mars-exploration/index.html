string_mistakes = 0
where_are_we = 1
for i in range(len(s)):
    if where_are_we == 1 or where_are_we == 3:
        if s[i] != "S":
            string_mistakes += 1
    elif where_are_we == 2:
        if s[i] != "O":
            string_mistakes += 1
    if where_are_we == 3:
        where_are_we = 1
    else:
        where_are_we += 1
print(string_mistakes)
